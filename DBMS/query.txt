//Account_tbl:
CREATE TABLE ACCOUNT_TBL (
    Account_Number VARCHAR2(20) CONSTRAINT Account_Number_pk PRIMARY KEY,
    Customer_Name VARCHAR2(15),
    Balance NUMBER(10,2)
);

//Transfer_tbl:
CREATE TABLE TRANSFER_TBL (
    Transaction_ID NUMBER(4) CONSTRAINT trans_id_pk PRIMARY KEY,
    Account_Number VARCHAR2(10),
    Beneficiary_Account_Number VARCHAR2(10),
    Transaction_Date DATE,
    Transaction_Amount NUMBER(10,2),
    CONSTRAINT acc_num_fk FOREIGN KEY (Account_Number) REFERENCES ACCOUNT_TBL(Account_Number),
    CONSTRAINT bene_acc_fk FOREIGN KEY (Beneficiary_Account_Number) REFERENCES ACCOUNT_TBL(Account_Number)
);


//Sequence:
create sequence transactionId_seq
		START WITH 1000
		MINVALUE 1000
		MAXVALUE 9999
		INCREMENT BY 1;
